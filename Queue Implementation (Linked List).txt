
#Linked List implementation of Queue:

class QueueEmptyError(Exception):
    pass
class Node:
    def __init__(self,val):
        self.data=val
        self.link=None

class Queue:
    def __init__(self):
        self.front=None
        self.rear=None
        self.size=0

    def is_empty(self):
        return self.front==None

    def sizeofQueue(self):
        return self.size

    def enqueue(self,data):
        temp=Node(data)
        if self.front==None:
            self.front=temp
        else:
            self.rear.link=temp
        self.rear=temp
        self.size+=1
    def dequeue(self):
        if self.is_empty():
            raise QueueEmptyError("Queue is empty.")
        x=self.front.data
        self.front=self.front.link
        self.size-=1
        return x
    def peek(self):
        if self.is_empty():
            raise QueueEmptyError("Queue is empty")
        return self.front.data
    def display(self):
        if self.is_empty():
            print("Queue is empty")
            return
        print("Queue is: ")
        p=self.front
        while p is not None:
            print(p.data,end=" ")
            p=p.link
        print()
